import { Button, Search, Checkbox, Radio, Input, Field, Link, Banner, Chip, Tooltip, Icon, Switch, Divider } from '@equinor/eds-core-react/next'
import { favorite, check, access_time, remove, check_circle, info_circle, warning_outlined } from '@equinor/eds-icons'
import { useEffect, useRef } from 'react'

export const PersistentTooltip = ({ title, placement = 'bottom' }) => {
  const ref = useRef(null)
  useEffect(() => {
    try { ref.current?.showPopover() } catch (e) {}
  }, [])
  return (
    <Tooltip ref={ref} title={title} placement={placement}>
      <span style={{ display: 'inline-block', width: '1px', height: '1px' }}>&#8203;</span>
    </Tooltip>
  )
}
